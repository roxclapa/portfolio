package teme.automation_course4;

import org.junit.Assert;
import org.junit.Before;
import org.junit.Test;
import org.openqa.selenium.By;
import teme.util.ta_utils.BaseTestClass;

/**
 * @author Roxana
 */

public class AlertsPracticeExercises extends BaseTestClass {
    
    public String yourName = "Roxana";

    @Before
    public void openAlertDemoURL() {
        driver.get("https://demo.automationtesting.in/Alerts.html");
    }

    @Test
    public void acceptTheAlertWillShow() {
        driver.findElement(By.xpath("//*[@href='#CancelTab']")).click();
        driver.findElement(By.cssSelector(".btn-primary")).click();
        driver.switchTo().alert().accept();

        Assert.assertEquals( "You pressed Ok", driver.findElement(By.id("demo")).getText());
    }

    @Test
    public void dismissTheAlertWillShow() {
        driver.findElement(By.xpath("//*[@href='#CancelTab']")).click();
        driver.findElement(By.xpath("//*[@id='CancelTab']/button")).click();
        driver.switchTo().alert().dismiss();

        Assert.assertEquals("You Pressed Cancel", driver.findElement(By.id("demo")).getText());
    }

    @Test
    public void getTextFromAlertWithOKAndCancelIsAsExpected() {
        driver.findElement(By.xpath("//*[@href='#CancelTab']")).click();
        driver.findElement(By.xpath("//*[@id='CancelTab']/button")).click();
        String alertMessage = driver.switchTo().alert().getText();

        Assert.assertEquals("The alert message is not correct", "Press a Button !", alertMessage);
    }

    @Test
    public void getTextFromAlertWithTextboxIsAsExpected() {
        driver.findElement(By.xpath("//*[@href='#Textbox']")).click();
        driver.findElement(By.cssSelector(".btn-info")).click();
        String alertMessage = driver.switchTo().alert().getText();

        Assert.assertEquals("The alert message is not correct", "Please enter your name", alertMessage);
    }

    @Test
    public void sendKeysToTheAlertWillShowTextName() {
        driver.findElement(By.xpath("//*[@href='#Textbox']")).click();
        driver.findElement(By.cssSelector(".btn-info")).click();
        driver.switchTo().alert().sendKeys(yourName);
        driver.switchTo().alert().accept();

        Assert.assertEquals("Hello Roxana How are you today", driver.findElement(By.id("demo1")).getText());
    }
}
