package teme.automation_course2;

import org.junit.Assert;
import org.junit.Test;
import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.ui.Select;
import teme.util.ta_utils.BaseTestClass;
import java.util.List;
                
/**
* @author Roxana
*/
                
public class DropdownSelect extends BaseTestClass {
/**
* Go to https://www.mediacollege.com/internet/samples/html/country-list.html
* Check that there are no countries starting with X
*/
                    
@Test
public void checkThatNoCountriesStartWithX() {
    driver.get("https://www.mediacollege.com/internet/samples/html/country-list.html");
    Select dropdown = new Select(driver.findElement(By.cssSelector("[name=country]")));
    List countries = dropdown.getOptions();
    for (WebElement country : countries) {
        Assert.assertFalse("Countries found!", country.getText().startsWith("X"));
        }
    }
}
