package teme.automation_course1;

import org.junit.Assert;
import org.junit.Test;
import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import teme.util.ta_utils.BaseTestClass;
import java.util.List;
                
/**
 * @author Roxana
 */
                
public class InterrogationHomework extends BaseTestClass {
                
/**
 * #Web Element Interrogation Exercises
 * Go to https://testare-manuala.locdejoacapentruitsti.com/blog/contact/
 * 1. Find the "Software Testing" button from the menu and identify it through its id.
Assert that its text value is correct.
 * 2. Find the ID for the Facebook button (upper right side). Extract its text.
 What would you expect the value to be?
 * 3. Find the Email input and assert on its tag name and that it's displayed.
 * 4. Find the WordPress link and assert that it is underlined. Check the css value in dev tools.
 * 5. Find all inputs - print their number on the screen.
 * 6. Find all the available selections from the dropdown - assert on their size, get the first ones text and print it on the screen,
 find the longest one and print it on the screen.
 * 7. Find all the social media links chaining find by's and assert on size
 * 8. Go back to exercise 2. Can you prove that the text visible in the DOM element is not visible?
 */
                
@Test
public void identifySoftwareTestingButtonThroughId(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    WebElement menuButton = driver.findElement(By.id("menu-item-115"));
                
    Assert.assertEquals("Software Testing", menuButton.getText());
    }
                
@Test
public void identifyFacebookButton(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    WebElement facebookButton = driver.findElement(By.id("menu-item-14"));
                        
    Assert.assertEquals("", facebookButton.getText());
    }
                
@Test
public void identifyEmailInput(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    WebElement emailInput = driver.findElement(By.name("your-email"));
                
    Assert.assertEquals("input", emailInput.getTagName());
    Assert.assertTrue("Email input is not displayed!", emailInput.isDisplayed());
    }
                
@Test
public void identifyWordpressLink(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    WebElement wordpressLink = driver.findElement(By.linkText("WordPress"));
                
    System.out.println(wordpressLink.getCssValue("text-decoration"));
                
    Assert.assertEquals("underline solid rgb(255, 255, 255)", wordpressLink.getCssValue("text-decoration"));
    }
                
@Test
public void identifyAllInputs(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    List allInputs = driver.findElements(By.tagName("input"));
    System.out.println("The list of inputs on the page contains " + allInputs.size() + " elements");
    }
                
@Test
public void identifyAllAvailableSelectionsFromDropdown(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");
    List dropdown = driver.findElements(By.tagName("option"));
                
    System.out.println("The number of available selections in the dropdown: " + dropdown.size());
                
    Assert.assertEquals(4, dropdown.size());
                
    System.out.println("The first option in the dropdown: " + dropdown.get(0).getText());
    System.out.println("Is the option selected? " + dropdown.get(0).isSelected());
    int longest = dropdown.get(0).getText().length();
    int index = 0;
    for (int i = 0; i < dropdown.size(); i++) {
        if (dropdown.get(i).getText().length() > longest) {
            longest = dropdown.get(i).getText().length();
            index = i;
            }
        }
    System.out.printf("The longest element is \"%s\" and it has the size of %d characters\n",
                        dropdown.get(index).getText(), longest);
    }
                
@Test
public void identifyAllSocialMediaLinks(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");      
    WebElement socialMediaMenu = driver.findElement(By.cssSelector(".screen-reader-text"))
                                    .findElement(By.xpath("//*[@id=\"menu-menu-1\"]"));
    List socialLinks = socialMediaMenu.findElements(By.tagName("li"));
    System.out.println("All the social media links: ");
    for (WebElement l : socialLinks) {          
        System.out.println("Link: " + l.getAttribute("textContent"));
        }
                
    Assert.assertEquals(4, socialLinks.size());
    }
                
@Test
public void verifyThatFacebookTextVisibleInDomIsHiddenForUsers(){
    driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/contact/");        
    WebElement facebookText = driver.findElement(By.xpath("//*[@id=\"menu-item-14\"]/a/span"));
    System.out.println("Get the text: " + facebookText.getText());
    System.out.println("Get the hidden text: " + facebookText.getAttribute("textContent"));
    }
