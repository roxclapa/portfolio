package teme.automation_course4;

import org.junit.Assert;
import org.junit.Before;
import org.junit.Test;
import org.openqa.selenium.By;
import org.openqa.selenium.WebElement;
import teme.util.ta_utils.BaseTestClass;
                
/**
* @author Roxana
*/
                
public class FramesExercise extends BaseTestClass {
                
@Before
public void getOnWebPage() {
    driver.get("https://demo.automationtesting.in/Frames.html");
}
                
@Test
public void single_iFrame()  {
    driver.switchTo().frame("singleframe");
    WebElement textInput = driver.findElement(By.cssSelector("[class='col-xs-6 col-xs-offset-5'] input"));
    textInput.sendKeys("Roxana testeaza");
    driver.switchTo().defaultContent();
    driver.findElement(By.cssSelector("[href='#Multiple']")).click();
    driver.findElement(By.cssSelector("[href='#Single']")).click();
    driver.switchTo().frame("singleframe");
                
    Assert.assertEquals("Roxana testeaza", textInput.getAttribute("value"));
    driver.switchTo().defaultContent();
    }
                
    @Test
    public void nested_iFrame() {
        driver.findElement(By.cssSelector("[href='#Multiple']")).click();
        driver.switchTo().frame(1);
        driver.switchTo().frame(0);
        WebElement textInput = driver.findElement(By.cssSelector("[class='col-xs-6 col-xs-offset-5'] input"));
        textInput.sendKeys("Roxana testeaza");
        driver.switchTo().parentFrame(); // <- just for fun
        driver.switchTo().defaultContent();
        driver.findElement(By.cssSelector("[href='#Single']")).click();
        driver.findElement(By.cssSelector("[href='#Multiple']")).click();
        driver.switchTo().frame(1);
        driver.switchTo().frame(0);
                
        Assert.assertEquals("Roxana testeaza", textInput.getAttribute("value"));
        driver.switchTo().defaultContent();
    }
}
