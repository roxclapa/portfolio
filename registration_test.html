package teme.automation_course2;
import org.junit.Assert;
import org.junit.Test;
import org.openqa.selenium.By;
import teme.util.ta_utils.BaseTestClass;
import static org.apache.commons.lang3.RandomStringUtils.randomAlphanumeric;

/**
* @author Roxana
*/

public class TestingRegister extends BaseTestClass {
    String numeRandom = randomAlphanumeric(7,21);

    @Test
    public void registerWithValidInputs(){
        driver.get("https://testare-manuala.locdejoacapentruitsti.com/blog/register/");
        driver.findElement(By.cssSelector("[id=username]")).sendKeys(numeRandom);
        driver.findElement(By.cssSelector("[name=email]")).sendKeys(numeRandom + "@gmail.com");
        driver.findElement(By.cssSelector("[id=passw1]")).sendKeys("Parolanounouta12");
        driver.findElement(By.cssSelector("[id=passw2]")).sendKeys("Parolanounouta12");
        driver.findElement(By.cssSelector("[name=register]")).submit();
        String mesaj = driver.findElement(By.cssSelector("[id=wppb_form_success_message]")).getText();
                    
        Assert.assertEquals("The account " + numeRandom + " has been successfully created!", mesaj);
    }
}
